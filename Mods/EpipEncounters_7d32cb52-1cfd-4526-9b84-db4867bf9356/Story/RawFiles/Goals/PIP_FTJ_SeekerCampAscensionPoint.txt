// Grant an Ascension point upon entering the seeker camp (Amadia Sanctuary). If this happens, the Ascension point gained from entering the act3 Academy from the base mod will not trigger. See the overridden CoS_AMER_AcademyAscPoint goal.

IF CharacterEnteredTrigger((CHARACTERGUID)_Char, TRIGGERGUID_S_FTJ_SW_XP_SeekerCamp_66a059dd-81b9-405c-bd53-305202b5b9e3)
AND
DB_IsPlayer(_Char)
THEN
DB_PIP_HasAscensionPointFromAmadiaSanctuary(1);
PROC_PIP_GrantAmadiaSanctuaryAscPoint();
GoalCompleted;

PROC PROC_PIP_GrantAmadiaSanctuaryAscPoint()
THEN
PROC_AMER_UI_Ascension_TotalPointsFound_CheckSet_FromSource(1, "PIP_AmadiaSanctuary");

PROC PROC_PIP_GrantAmadiaSanctuaryAscPoint()
AND
DB_IsPlayer(_Player)
THEN
PROC_AMER_UI_Ascension_PointsRemaining_TryAdd_FromSource(_Player, 1, "PIP_AmadiaSanctuary");
